library(huxtable)
library(tidyverse)

df_aov <- data.frame("Source of Variation" = c("Total:", "• Varieties", "• Error"),
                     "Degrees of Freedom, df" = c(9, 1, 8),
                     "Sum of Squares, SS" = c("64.5", "22.5", "42.0"),
                     "Mean Square, MS" = c("", "22.50", "5.25"),
                     "Observed F" = c("", 4.29, ""),
                     "Required F, 10%" = c("", 3.46, ""),
                     "Required F, 5%" = c("", 5.32, ""),
                     check.names = FALSE
                     )

df_aov_ok <- df_aov                                                      |> 
             as_hux()                                                    |>
             set_contents(1, 6:7, c("10%", "5%"))                        |>
             insert_row("", "", "", "", "", "Required F", "", after = 0) |> 
             merge_cells(1, 6:7)                                         |>
             merge_cells(1:2, 1)                                         |> # Merge "Source of Variation"
             merge_cells(1:2, 2)                                         |> # Merge "Degrees of Freedom, df"
             merge_cells(1:2, 3)                                         |> # Merge "Sum of Squares, SS"  
             merge_cells(1:2, 4)                                         |> # Merge "Mean Square, MS"  
             merge_cells(1:2, 5)                                         |> # Merge "Observed F"
             theme_article()                                             |>
             set_tb_padding(1)                                           |>
             set_align(1, everywhere, "center")                          |>
             set_bold(1, everywhere)                                     |>
             set_width(1)                                                |> 
             # set_font_size(13)                                           |> 
             set_lr_padding(1)                                           |> 
             set_col_width(c(0.15, 0.15, 0.15, 0.15, 0.2, 0.1, 0.1))     |>
             set_position("center")

df_aov_ok[1, 1] <- "Source of Variation"
df_aov_ok[1, 2] <- "Degrees of Freedom, df partial $\\sqrt {{a^2} + {b^2}}$" # double backslash
df_aov_ok[1, 3] <- "Sum of Squares, SS"
df_aov_ok[1, 4] <- "Mean Square, MS"
df_aov_ok[1, 5] <- "Observed F"

huxtable::align(df_aov_ok)[2, 6:7] <- "center" # chỉnh 10% và 5% về center

huxtable::align(df_aov_ok)[3:5, 2:7] <- "right"
